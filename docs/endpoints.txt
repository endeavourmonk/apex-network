1. Users:
    - GET: /users?username={username}
    - POST: /users
    - PUT: /users/{userID}
    - PATCH: /users/{userID}
    - DELETE: /users/{userID}

2. Posts:
    - GET: /posts?userID={userID}
    - POST: /posts

    - PATCH: /posts/{postID}
    - DELETE: /posts/{postID}

3. Connections:

    - POST: /connections
